# The three countries with the most numbers of deaths are US, Brazil and India
  1. Use transition_reveal to make a plot revealing the number of deaths in the three countries with the most numbers of deaths
library(dplyr)
library(ggplot2)
library(gganimate)
library(lubridate)

df$Date_reported = as.Date(df$Date_reported, format = "%m/%d/%Y")

df_top3 <- df %>%
  filter(Country %in% c("United States of America", "Brazil", "India")) %>%
  arrange(Country, Date_reported) %>%
  distinct(Country, Date_reported, .keep_all = TRUE)   # <-- SUPER IMPORTANT

plot_deaths <- ggplot(df_top3, aes(x = Date_reported,
                                   y = Cumulative_deaths,
                                   color = Country,
                                   group = Country)) +
  geom_line(size = 1) +
  transition_reveal(Date_reported) +
  labs(title = "Cumulative Deaths by Covid-19 in Top 3 Countries",
       x = "Date", y = "Cumulative Deaths")
animate(plot_deaths, nframes = 100, fps = 10)

dff <- df %>%

  1. Use transition_reveal to make a plot revealing the number of deaths per cases of the US, Italy and Mexico.
dff_selected <- dff %>% filter(Country %in% c("United States of America", "Italy", "Mexico"))
plot_death_per_cases <- ggplot(dff_selected, aes(x = Date_reported, y = death_per_cases, color = Country)) +
  geom_line(size = 1) +
  labs(title = "Death per Cases by Covid-19 in Selected Countries",
       x = "Date",
       y = "Death per Cases",
       caption = "Data Source: WHO") +
  transition_reveal(Date_reported)
animate(plot_death_per_cases, nframes = 100, fps = 10)
